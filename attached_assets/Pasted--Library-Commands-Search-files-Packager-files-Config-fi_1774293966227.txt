✅ 4. Registration Form (Frontend)

Here’s a simple React form (you can use in your app):

import { useState } from "react";
import { supabase } from "./supabaseClient";

export default function Register() {
  const [form, setForm] = useState({});

  const handleSubmit = async (e) => {
    e.preventDefault();

    const { data, error } = await supabase
      .from("learners")
      .insert([form]);

    if (error) alert(error.message);
    else alert("Registered successfully!");
  };

  return (
    <div className="container">
      <img src="/logo.png" alt="logo" className="logo" />

      <form onSubmit={handleSubmit}>
        <input placeholder="Name" onChange={e => setForm({...form, name: e.target.value})} />
        <input placeholder="Surname" onChange={e => setForm({...form, surname: e.target.value})} />
        <input placeholder="Email" onChange={e => setForm({...form, email: e.target.value})} />
        <input placeholder="Phone" onChange={e => setForm({...form, phone: e.target.value})} />

        <select onChange={e => setForm({...form, province: e.target.value})}>
          <option>Western Cape</option>
          <option>Gauteng</option>
          <option>KZN</option>
        </select>

        <button type="submit">Register</button>
      </form>
    </div>
  );
}
