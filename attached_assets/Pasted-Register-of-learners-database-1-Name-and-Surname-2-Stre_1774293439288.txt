✅ 1. Database Structure (Supabase)

Create these tables in Supabase:

📌 learners
create table learners (
  id uuid primary key default uuid_generate_v4(),
  name text,

  street_address text,
  email text,
  phone text,
  parent_details text,
  province text,

  subjects text[],

  gender text,

  municipality text,

✅ 3. API (Supabase auto API)

Supabase already gives you an API automatically.

Example: Fetch schools
const { data, error } = await supabase
  .from('schools')
  .select('*')
Filter by province:
const { data } = await supabase
  .from('schools')
  .select('*')
  .eq('province', 'Western Cape')
✅ 4. Registration Form (Frontend)

Here’s a simple React form (you can use in your app):

import { useState } from "react";
import { supabase } from "./supabaseClient";

export default function Register() {
  const [form, setForm] = useState({});

  const handleSubmit = async (e) => {
    e.preventDefault();

    const { data, error } = await supabase
      .from("learners")
      .insert([form]);

    if (error) alert(error.message);
    else alert("Registered successfully!");
  };

  return (
    <div className="container">
      <img src="/logo.png" alt="logo" className="logo" />

      <form onSubmit={handleSubmit}>
        <input placeholder="Name" onChange={e => setForm({...form, name: e.target.value})} />
        <input placeholder="Surname" onChange={e => setForm({...form, surname: e.target.value})} />
        <input placeholder="Email" onChange={e => setForm({...form, email: e.target.value})} />
        <input placeholder="Phone" onChange={e => setForm({...form, phone: e.target.value})} />

        <select onChange={e => setForm({...form, province: e.target.value})}>
          <option>Western Cape</option>
          <option>Gauteng</option>
          <option>KZN</option>
        </select>

        <button type="submit">Register</button>
      </form>
    </div>
  );
}
